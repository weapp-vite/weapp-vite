## 示例 {#lifecycle-examples}

```vue [TypeScript]
<script setup lang="ts">
import {
  onMounted,
  onPageScroll,
  onRouteDone,
  onShareAppMessage,
  onShow,
  ref,
} from 'wevu'

const scrollTop = ref(0)

onShow(() => {
  console.log('page show')
})

onMounted(() => {
  console.log('mounted')
})

onPageScroll((e) => {
  scrollTop.value = e.scrollTop
})

onRouteDone(() => {
  console.log('route done')
})

onShareAppMessage(() => ({
  title: 'Wevu 页面分享',
  path: '/pages/index/index',
}))
</script>

<template>
  <view>scrollTop: {{ scrollTop }}</view>
</template>
```
